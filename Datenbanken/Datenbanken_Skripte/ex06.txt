CREATE OR REPLACE VIEW ex06.autoperson
AS SELECT
    p.name,
    p.vorname,
    a.marke,
    a.typ,
    a.baujahr
FROM
    ex06.auto a
INNER JOIN ex06.fahrzeughalter f ON
    a.anr = f.anr
    AND a.anr = f.anr
INNER JOIN ex06.person p ON
    f.pnr = p.pnr
    AND f.pnr = p.pnr;
   
CREATE OR REPLACE VIEW ex06.auto_allperson
AS SELECT
    p.name,
    p.vorname,
    a.marke,
    a.typ,
    a.baujahr
FROM
    ex06.auto a
RIGHT JOIN ex06.fahrzeughalter f ON
    a.anr = f.anr
    AND a.anr = f.anr
RIGHT JOIN ex06.person p ON
    f.pnr = p.pnr
    AND f.pnr = p.pnr;

CREATE OR REPLACE VIEW ex06.person_allauto
AS SELECT
    a.marke,
    a.typ,
    a.baujahr,
	p.name,
    p.vorname
FROM
    ex06.person p
RIGHT JOIN ex06.fahrzeughalter f ON
    p.pnr = f.pnr
    AND p.pnr = f.pnr
RIGHT JOIN ex06.auto a ON
    f.anr = a.anr
    AND f.anr = a.anr;
   
CREATE OR REPLACE VIEW ex06.allperson_nocar 
AS SELECT
    p.name,
    p.vorname
FROM
    ex06.auto a
RIGHT JOIN ex06.fahrzeughalter f ON
    a.anr = f.anr
    AND a.anr = f.anr
RIGHT JOIN ex06.person p ON
    f.pnr = p.pnr
    AND f.pnr = p.pnr
WHERE a.baujahr IS NULL;

CREATE OR REPLACE VIEW ex06.allcar_noperson
AS SELECT
    a.marke,
    a.typ,
    a.baujahr
FROM
    ex06.person p
RIGHT JOIN ex06.fahrzeughalter f ON
    p.pnr = f.pnr
    AND p.pnr = f.pnr
RIGHT JOIN ex06.auto a ON
    f.anr = a.anr
    AND f.anr = a.anr
WHERE p.vorname IS NULL;
