CREATE OR REPLACE VIEW ex04.autoperson
AS SELECT
	a.pnr,
	p.name,
	p.vorname,

CREATE OR REPLACE VIEW ex04.auto_allperson
AS SELECT
    p.pnr,
    p.name,
    p.vorname,
    a.marke,
    a.typ,
    a.baujahr
FROM
    ex04.auto a
RIGHT JOIN ex04.person p ON
    a.pnr = p.pnr;

CREATE OR REPLACE VIEW ex04.person_no_car
AS SELECT
    p.pnr,
    p.name,
    p.vorname,
    a.marke,
    a.typ,
    a.baujahr
FROM
    ex04.auto a
RIGHT JOIN ex04.person p ON
    a.pnr = p.pnr
WHERE 
	a.baujahr IS NULL;
